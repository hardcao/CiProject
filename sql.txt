 1 create table T_SUBSCRIBECONFIRMRECORD (FID varchar(10) PRIMARY KEY,
					   FPROJECTID VARCHAR(10), 
					   FUSERID VARCHAR(10), 
					   FBANKID VARCHAR(10), 
					   FAMOUNT VARCHAR(10), 
					   FLEVERRATIO VARCHAR(10), 
					   FLEVERAMOUNT VARCHAR(10), 
					   FCONFIRMAMOUNT VARCHAR(10),
					   FLEVERCONFIRMAMOUNT VARCHAR(10));

 2 create table T_PAYRECORD(FID varchar(10) PRIMARY KEY,
			    FSUBSCRIBECONFIGRMRECORDID VARCHAR(10), 
			    FPAYTIMES INT, 
			    FPAYDATE DATE, 
			    FPAYAMOUNT VARCHAR(10), 
			    FLEVERAMOUNT TIMESTAMP );

3 create table T_BONUSRECORD(FID varchar(10) PRIMARY KEY,
			    FSUBSCRIBECONFIGRMRECORDID VARCHAR(10), 
			    FBONUSTIMES INT, 
			    FBONUSDATE DATE, 
			    FBONUSAMOUNT VARCHAR(10), 
			    FCREATETIME TIMESTAMP );
